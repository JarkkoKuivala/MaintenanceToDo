CREATE TABLE Devices 
(
    id INTEGER PRIMARY KEY,
    device_name TEXT,
    model TEXT,
    user_id INTEGER REFERENCES Users(id)
);
CREATE TABLE Tasks 
(
    id INTEGER PRIMARY KEY,
    task TEXT,
    deadline TEXT,
    completed TEXT,
    repetetive TEXT,
    take_note TEXT,
    publishing VARCHAR(50),
    users_id INTEGER REFERENCES Users(id),
    devices_id INTEGER REFERENCES  Devices(id)
);
CREATE TABLE Users 
(
    id INTEGER PRIMARY KEY,
    name TEXT
);
